function uploadFile(kode,index){ let input = document.getElementById("file"+index); let file = input.files[0]; if(!file){ alert("Pilih file dulu"); return; } let reader = new FileReader(); reader.onload=function(e){ // SIMPAN DOKUMEN let dokumen = JSON.parse(localStorage.getItem("dokumenSPJ")) || []; dokumen = dokumen.filter(x=>x.kode!=kode); dokumen.push({ kode:kode, nama:file.name, url:e.target.result }); localStorage.setItem( "dokumenSPJ", JSON.stringify(dokumen) ); // UPDATE STATUS SPJ let spj = JSON.parse(localStorage.getItem("spj")) || []; spj = spj.map(x=>{ if(x.kode==kode){ return { ...x, status:"Selesai", dokumen:true }; } return x; }); localStorage.setItem( "spj", JSON.stringify(spj) ); alert("SPJ berhasil diverifikasi"); location.reload(); } reader.readAsDataURL(file); }
