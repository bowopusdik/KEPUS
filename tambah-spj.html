document.getElementById("formSPJ").addEventListener("submit", function(e){ e.preventDefault(); let data = JSON.parse(localStorage.getItem("dataSPJ")) || []; const nomorSurat = document.getElementById("nomorSurat").value; const pokja = document.getElementById("pokja").value; const nilaiSPJ = document.getElementById("nilaiSPJ").value; const status = document.getElementById("status").value; const newData = { id: Date.now(), nomorSurat: nomorSurat, pokja: pokja, nilaiSPJ: Number(nilaiSPJ), status: status }; data.push(newData); localStorage.setItem("dataSPJ", JSON.stringify(data)); alert("SPJ berhasil ditambahkan!"); window.location.href = "data-spj.html"; });
