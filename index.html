let spmData = [ { id: 1, spm: "SPM-001/2026", total: "5.000.000", drpp: "DRPP-01", spby: "SPBY-01", kwitansi: "KW-01", st: "ST-01", spj: ["spj_001.pdf"], status: "TERVERIFIKASI" }, { id: 2, spm: "SPM-002/2026", total: "3.500.000", drpp: "DRPP-02", spby: "SPBY-02", kwitansi: "KW-02", st: "ST-02", spj: [], status: "BELUM UPLOAD" } ]; // ================= SEARCH ================= function searchData(keyword){ keyword = keyword.toLowerCase(); return spmData.filter(item => item.spm.toLowerCase().includes(keyword) || item.total.includes(keyword) || item.drpp.toLowerCase().includes(keyword) || item.spby.toLowerCase().includes(keyword) || item.kwitansi.toLowerCase().includes(keyword) ); } // ================= RENDER TABLE ================= function renderTable(data){ const table = document.querySelector("table"); let rows = ` View Status SPM Total SPM DRPP SPBY KWITANSI SURAT TUGAS SPJ Aksi `; data.forEach(item => { rows += ` ${getStatus(item)} ${item.spm} ${item.total} ${item.drpp} ${item.spby} ${item.kwitansi} ${item.st} Detail `; }); table.innerHTML = rows; } // ================= STATUS ================= function getStatus(item){ if(item.spj.length === 0){ return '🔴 BELUM UPLOAD'; } else if(item.status === "REVISI"){ return '🟡 REVISI'; } else { return '🟢 TERVERIFIKASI'; } } // ================= UPLOAD SPJ ================= function uploadSPJ(id){ let file = prompt("Masukkan nama file SPJ (contoh: spj_003.pdf)"); if(file){ let item = spmData.find(d => d.id === id); item.spj.push(file); item.status = "TERVERIFIKASI"; alert("SPJ berhasil diupload"); renderTable(spmData); } } // ================= DELETE SPJ ================= function deleteSPJ(id){ let item = spmData.find(d => d.id === id); if(confirm("Hapus SPJ?")){ item.spj = []; item.status = "BELUM UPLOAD"; renderTable(spmData); } } // ================= MODAL VIEW ================= function openModal(id){ let item = spmData.find(d => d.id === id); alert( "SPM: " + item.spm + "\nTotal: " + item.total + "\nDRPP: " + item.drpp + "\nSPBY: " + item.spby + "\nKWITANSI: " + item.kwitansi + "\nST: " + item.st + "\nSPJ: " + (item.spj.join(", ") || "Belum upload") ); } // ================= SEARCH INPUT ================= document.addEventListener("DOMContentLoaded", function(){ const input = document.querySelector("input"); input.addEventListener("keyup", function(){ let result = searchData(this.value); renderTable(result); }); renderTable(spmData); });
